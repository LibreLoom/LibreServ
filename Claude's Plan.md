## Project Overview

**LibreServ** is a self-hosting platform that makes running your own server as simple as using a smartphone. It provides:

- **Single binary server** written in Go
- **Web-based UI** for all management (no terminal required)
- **Three app types**: Builtin (catalog), Custom (user-uploaded), External (user-deployed)
- **Automatic updates** with rollback capability
- **Unified monitoring** for all apps
- **Backup system** with scheduling
- **Automatic HTTPS** via Caddy

### Core Philosophy

- No terminal required for 95% of users
- Everything should "just work" out of the box
- Every action should be reversible
- Plain language, not technical jargon

```go
package api

import (
    "net/http"
    "github.com/go-chi/chi/v5"
    "github.com/go-chi/chi/v5/middleware"
)

type Server struct {
    router chi.Router
    addr   string
}

func NewServer(addr string) *Server {
    r := chi.NewRouter()
    
    // Global middleware
    r.Use(middleware.Logger)
    r.Use(middleware.Recoverer)
    r.Use(middleware.RequestID)
    r.Use(CORSMiddleware())
    
    return &Server{
        router: r,
        addr:   addr,
    }
}

func (s *Server) Start() error {
    return http.ListenAndServe(s.addr, s.router)
}

func (s *Server) SetupRoutes(handlers *Handlers) {
    // Public routes
    s.router.Group(func(r chi.Router) {
        r.Post("/api/auth/login", handlers.Login)
        r.Get("/api/health", handlers.HealthCheck)
    })
    
    // Protected routes
    s.router.Group(func(r chi.Router) {
        r.Use(AuthMiddleware)
        
        // Apps
        r.Get("/api/apps", handlers.ListApps)
        r.Get("/api/apps/{id}", handlers.GetApp)
        r.Post("/api/apps/{id}/start", handlers.StartApp)
        r.Post("/api/apps/{id}/stop", handlers.StopApp)
        
        // More routes...
    })
}
```

```go
type Route struct {
    Subdomain  string
    Backend    string
    SSL        bool
}

func (cm *CaddyManager) AddRoute(subdomain, backend string) error {
    // 1. Add route to in-memory map
    cm.routes[subdomain] = &Route{
        Subdomain: subdomain,
        Backend:   backend,
        SSL:       true,
    }
    
    // 2. Regenerate Caddyfile
    err := cm.GenerateCaddyfile()
    
    // 3. Reload Caddy
    return cm.ReloadCaddy()
}

func (cm *CaddyManager) GenerateCaddyfile() error {
    // Use template to generate Caddyfile
    tmpl := `
{{range .Routes}}
{{.Subdomain}}.{$DOMAIN} {
    reverse_proxy {{.Backend}}
    {{if .SSL}}
    tls {
        on_demand
    }
    {{end}}
}
{{end}}
`
    // Write to file
}

func (cm *CaddyManager) ReloadCaddy() error {
    // Send reload signal to Caddy
    // OR use Caddy API
    return exec.Command("caddy", "reload", "--config", cm.configPath).Run()
}
```

```go
package updates

import (
    "context"
    "fmt"
    "time"
)

type UpdateManager struct {
    docker     *docker.Client
    db         *database.DB
    backup     *storage.BackupService
    health     *monitoring.HealthChecker
    logger     *log.Logger
}

type UpdateResult struct {
    Success       bool
    AppID         string
    OldVersions   []ImageVersion
    NewVersions   []ImageVersion
    BackupID      string
    Duration      time.Duration
    Error         error
    RolledBack    bool
}

func (m *UpdateManager) UpdateApp(ctx context.Context, appID string, opts UpdateOptions) (*UpdateResult, error) {
    result := &UpdateResult{
        AppID:     appID,
        StartTime: time.Now(),
    }
    
    // Load app
    app, err := m.db.GetApp(appID)
    if err != nil {
        return nil, fmt.Errorf("failed to load app: %w", err)
    }
    
    // Verify app is managed
    if app.Type == "external" {
        return nil, fmt.Errorf("cannot update external apps")
    }
    
    // Step 1: Record current versions
    currentVersions, err := m.getCurrentVersions(app)
    if err != nil {
        return nil, fmt.Errorf("failed to get current versions: %w", err)
    }
    result.OldVersions = currentVersions
    
    // Step 2: Create backup
    if opts.BackupBeforeUpdate {
        m.logger.Infof("Creating backup for %s", appID)
        backupID, err := m.backup.CreateSnapshot(appID)
        if err != nil {
            return nil, fmt.Errorf("backup failed: %w", err)
        }
        result.BackupID = backupID
    }
    
    // Step 3: Set app status
    m.db.UpdateAppStatus(appID, "updating")
    
    // Step 4: Pull new images
    m.logger.Infof("Pulling new images for %s", appID)
    if err := m.docker.ComposePull(app.Path); err != nil {
        return m.handleUpdateFailure(result, err, "pull failed")
    }
    
    // Step 5: Stop app
    m.logger.Infof("Stopping %s", appID)
    if err := m.docker.ComposeDown(app.Path); err != nil {
        return m.handleUpdateFailure(result, err, "stop failed")
    }
    
    // Step 6: Start with new images
    m.logger.Infof("Starting %s with new images", appID)
    if err := m.docker.ComposeUp(app.Path); err != nil {
        return m.handleUpdateFailure(result, err, "start failed")
    }
    
    // Step 7: Wait for containers to be healthy
    m.logger.Infof("Waiting for %s to be healthy", appID)
    if err := m.waitForHealthy(ctx, appID, 2*time.Minute); err != nil {
        return m.handleUpdateFailure(result, err, "health check failed")
    }
    
    // Step 8: Verify app is working
    if opts.HealthCheckAfter {
        if err := m.health.CheckApp(appID); err != nil {
            return m.handleUpdateFailure(result, err, "verification failed")
        }
    }
    
    // Success!
    result.Success = true
    result.Duration = time.Since(result.StartTime)
    result.NewVersions, _ = m.getCurrentVersions(app)
    
    m.db.UpdateAppStatus(appID, "running")
    m.saveUpdateRecord(result)
    
    return result, nil
}

func (m *UpdateManager) handleUpdateFailure(result *UpdateResult, err error, stage string) (*UpdateResult, error) {
    result.Success = false
    result.Error = fmt.Errorf("%s: %w", stage, err)
    
    // Attempt rollback if backup exists
    if result.BackupID != "" {
        m.logger.Errorf("Update failed at %s, attempting rollback", stage)
        if rbErr := m.backup.RestoreSnapshot(result.BackupID); rbErr != nil {
            m.logger.Errorf("Rollback failed: %v", rbErr)
        } else {
            result.RolledBack = true
            m.logger.Infof("Successfully rolled back to backup %s", result.BackupID)
        }
    }
    
    m.db.UpdateAppStatus(result.AppID, "error")
    m.saveUpdateRecord(result)
    
    return result, result.Error
}
```

```go
package updates

import (
    "github.com/robfig/cron/v3"
)

type Scheduler struct {
    cron    *cron.Cron
    manager *UpdateManager
    db      *database.DB
}

func NewScheduler(manager *UpdateManager, db *database.DB) *Scheduler {
    return &Scheduler{
        cron:    cron.New(),
        manager: manager,
        db:      db,
    }
}

func (s *Scheduler) Start() {
    // Check for scheduled updates every hour
    s.cron.AddFunc("0 * * * *", s.checkScheduledUpdates)
    s.cron.Start()
}

func (s *Scheduler) checkScheduledUpdates() {
    apps := s.db.ListAppsWithAutoUpdate()
    
    for _, app := range apps {
        config := loadAppConfig(app.Path)
        
        // Check if it's time to update
        if s.shouldUpdate(app, config) {
            go s.manager.UpdateApp(context.Background(), app.ID, UpdateOptions{
                BackupBeforeUpdate: config.Updates.Safety.BackupBeforeUpdate,
                HealthCheckAfter:   config.Updates.Safety.HealthCheckAfter,
                RollbackOnFailure:  config.Updates.Safety.RollbackOnFailure,
            })
        }
    }
}

func (s *Scheduler) shouldUpdate(app *App, config *AppConfig) bool {
    // Check if within maintenance window
    if !s.isInMaintenanceWindow(config.Updates.Safety.MaintenanceWindow) {
        return false
    }
    
    // Check update strategy
    if config.Updates.Strategy != "auto" && config.Updates.Strategy != "scheduled" {
        return false
    }
    
    // Check if updates are available
    return app.UpdateAvailable
}
```

### Phase 6: Build System (Week 17)

#### 6.1 Multi-Architecture Builds

**Script**: `scripts/build.sh`

```bash
#!/bin/bash
set -e

VERSION=${1:-"dev"}
BUILD_DIR="bin"
PLATFORMS=(
  "linux/amd64"
  "linux/arm64"
  "linux/arm/v7"
)

echo "Building LibreServ v${VERSION}"

# Build frontend
echo "Building frontend..."
cd web
npm run build
cd ..

# Build for each platform
for PLATFORM in "${PLATFORMS[@]}"; do
  GOOS=$(echo $PLATFORM | cut -d'/' -f1)
  GOARCH=$(echo $PLATFORM | cut -d'/' -f2)
  GOARM=$(echo $PLATFORM | cut -d'/' -f3 | sed 's/v//')
  
  OUTPUT="${BUILD_DIR}/libreserv-${GOOS}-${GOARCH}"
  if [ -n "$GOARM" ]; then
    OUTPUT="${OUTPUT}v${GOARM}"
  fi
  
  echo "Building for ${PLATFORM}..."
  
  GOOS=$GOOS GOARCH=$GOARCH GOARM=$GOARM go build \
    -ldflags="-X main.Version=${VERSION} -X main.BuildTime=$(date -u +%Y-%m-%dT%H:%M:%SZ)" \
    -o $OUTPUT \
    ./cmd/libreserv
  
  # Compress binary
  upx --best --lzma $OUTPUT 2>/dev/null || true
  
  echo "✓ Built: $OUTPUT"
done

echo "Build complete!"
ls -lh $BUILD_DIR/
```

```bash
#!/bin/bash

DATA_DIR="/var/lib/libreserv"

case $ARCH in
  x86_64)
    ARCH="amd64"
    ;;
  aarch64)
    ARCH="arm64"
    ;;
  armv7l)
    ARCH="armv7"
    ;;
  *)
    echo "Unsupported architecture: $ARCH"
    exit 1
    ;;
esac

# Check if running as root
if [ "$EUID" -ne 0 ]; then
  echo "Please run as root or with sudo"
  exit 1
fi

# Check dependencies
echo "Checking dependencies..."
command -v docker >/dev/null 2>&1 || {
  echo "Docker is required but not installed."
  echo "Install Docker: https://docs.docker.com/engine/install/"
  exit 1
}

# Download LibreServ
echo "Downloading LibreServ for $ARCH..."
DOWNLOAD_URL="https://github.com/yourusername/libreserv/releases/latest/download/libreserv-linux-${ARCH}"
curl -L -o /tmp/libreserv $DOWNLOAD_URL

# Install binary
echo "Installing LibreServ..."
chmod +x /tmp/libreserv
mv /tmp/libreserv $INSTALL_DIR/libreserv

# Create directories
mkdir -p $CONFIG_DIR
mkdir -p $DATA_DIR/{apps,backups,logs,config}

# Create default config
cat > $CONFIG_DIR/config.yaml <<EOF
server:
  host: "0.0.0.0"
  port: 8080
  
database:
  path: "$DATA_DIR/config/libreserv.db"
  
apps:
  data_path: "$DATA_DIR/apps"
  
docker:
  socket: "unix:///var/run/docker.sock"
  
logging:
  level: info
  path: "$DATA_DIR/logs"
EOF

# Create systemd service
cat > $SERVICE_FILE <<EOF
[Unit]
Description=LibreServ - Simple Self-Hosting Platform

Requires=docker.service

[Service]
Type=simple
User=root
ExecStart=$INSTALL_DIR/libreserv serve --config $CONFIG_DIR/config.yaml

[Install]
WantedBy=multi-user.target
EOF

# Create Docker network
docker network create libreserv 2>/dev/null || true

# Enable and start service
systemctl daemon-reload
systemctl enable libreserv
systemctl start libreserv

echo ""
echo "✓ LibreServ installed successfully!"
echo ""
echo "Next steps:"
echo "1. Wait a few seconds for the service to start"
echo "2. Access LibreServ at: http://$(hostname -I | awk '{print $1}'):8080"
echo "3. Check status: systemctl status libreserv"
echo "4. View logs: journalctl -u libreserv -f"
echo ""
echo "First-time login credentials will be displayed in the logs."
```

## Best Practices

### Code Quality

#### Go Best Practices

1. **Error Handling**

```go
// Good: Return wrapped errors with context
func InstallApp(id string) error {
    if err := validateAppID(id); err != nil {
        return fmt.Errorf("invalid app ID %s: %w", id, err)
    }
    // ...
}

// Bad: Ignore or swallow errors
func InstallApp(id string) error {
    validateAppID(id) // Error ignored!
    // ...
}
```

1. **Context Usage**

```go
// Good: Pass context for cancellation
func (d *DockerClient) PullImage(ctx context.Context, image string) error {
    resp, err := d.cli.ImagePull(ctx, image, types.ImagePullOptions{})
    // ...
}

// Use context timeouts
ctx, cancel := context.WithTimeout(context.Background(), 5*time.Minute)
defer cancel()
```

1. **Concurrency**

```go
// Good: Use sync.WaitGroup for goroutines
var wg sync.WaitGroup
for _, app := range apps {
    wg.Add(1)
    go func(a *App) {
        defer wg.Done()
        checkHealth(a)
    }(app)
}
wg.Wait()

// Good: Use channels for communication
results := make(chan HealthResult, len(apps))
for _, app := range apps {
    go func(a *App) {
        results <- checkHealth(a)
    }(app)
}
```

1. **Interfaces**

```go
// Good: Small, focused interfaces
type AppInstaller interface {
    Install(appID string, config map[string]interface{}) error
}

type AppManager interface {
    Start(appID string) error
    Stop(appID string) error
    Restart(appID string) error
}

// Bad: Large, monolithic interface
type AppService interface {
    Install(appID string, config map[string]interface{}) error
    Start(appID string) error
    Stop(appID string) error
    Update(appID string) error
    Backup(appID string) error
    // ... 20 more methods
}
```

#### React Best Practices

1. **Component Structure**

```javascript
// Good: Small, focused components
function AppCard({ app }) {
  const { start, stop } = useAppActions(app.id);
  
  return (
    <Card>
      <AppHeader app={app} />
      <AppMetrics metrics={app.metrics} />
```

1. **Custom Hooks**

```javascript
// Good: Extract reusable logic
function useAppActions(appId) {
  const queryClient = useQueryClient();
  
  const startMutation = useMutation({
    mutationFn: () => appsAPI.start(appId),
```

1. **Error Boundaries**

```javascript
class ErrorBoundary extends React.Component {
  state = { hasError: false };
  
  static getDerivedStateFromError(error) {
    return { hasError: true };
  }
  
  componentDidCatch(error, errorInfo) {
    console.error('Error caught:', error, errorInfo);
  }
  
  render() {
    if (this.state.hasError) {
      return <ErrorFallback />;
    }
    return this.props.children;
  }
}
```

### Security Considerations

1. **Input Validation**

```go
// Always validate user input
func validateAppConfig(config map[string]interface{}) error {
    if name, ok := config["name"].(string); !ok || name == "" {
        return errors.New("name is required")
    }
    
    // Sanitize paths
    if path, ok := config["data_path"].(string); ok {
        if strings.Contains(path, "..") {
            return errors.New("invalid path: cannot contain '..'")
        }
    }
    
    return nil
}
```

1. **SQL Injection Prevention**

```go
// Good: Use parameterized queries
func (r *AppRepository) GetByID(id string) (*App, error) {
    var app App
    err := r.db.Get(&app, "SELECT * FROM apps WHERE id = ?", id)
    return &app, err
}

// Bad: String concatenation
func (r *AppRepository) GetByID(id string) (*App, error) {
    query := "SELECT * FROM apps WHERE id = '" + id + "'" // NEVER DO THIS
    // ...
}
```

1. **Password Security**

```go
// Always hash passwords
import "golang.org/x/crypto/bcrypt"

func hashPassword(password string) (string, error) {
    hash, err := bcrypt.GenerateFromPassword([]byte(password), bcrypt.DefaultCost)
    return string(hash), err
}

// Verify with constant-time comparison
func verifyPassword(hash, password string) error {
    return bcrypt.CompareHashAndPassword([]byte(hash), []byte(password))
}
```

```go
jwtSecret := make([]byte, 32)
```
